âœ… Simple PUT Example (Rest Assured Java)
import static io.restassured.RestAssured.*;
import static org.hamcrest.Matchers.*;

public class PutExampleTest {

    @Test
    public void updateUser_usingPut() {

        String requestBody = "{\n" +
                "  \"name\": \"Abhinav Tiwari\",\n" +
                "  \"email\": \"abhinav.tiwari@company.com\",\n" +
                "  \"role\": \"QA Lead\"\n" +
                "}";

        given()
            .baseUri("https://api.example.com")
            .contentType("application/json")
            .pathParam("id", 101)
            .body(requestBody)
        .when()
            .put("/users/{id}")
        .then()
            .statusCode(200)
            .body("name", equalTo("Abhinav Tiwari"))   // Hamcrest matches. 
            .body("role", equalTo("QA Lead"));
    }
}
